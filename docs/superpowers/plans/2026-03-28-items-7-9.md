```c
/* Block all pending write/critical commands (permit_write was disabled).
 * Changes PENDING → BLOCKED for entries with safety > CMD_SAFE.
 * Returns number blocked. */

```

- [ ] **Step 2: Add implementation to chat_approval.c**

After the `chat_approval_unblock_all` function (after line 141), add:

```c
int chat_approval_block_pending_writes(ApprovalQueue *q)
{
    int n = 0;
    for (int i = 0; i < q->count; i++) {
        if (q->entries[i].status == APPROVE_PENDING &&
            q->entries[i].safety > CMD_SAFE) {
            q->entries[i].status = APPROVE_BLOCKED;
            n++;
        }
    }
    return n;
}
```

```c
        case IDC_CHAT_PERMIT:
            if (d) {
                d->permit_write = !d->permit_write;
                InvalidateRect(d->hPermitBtn, NULL, TRUE);
                if (d->permit_write) {
                    /* Enabling: unblock all blocked commands */
                    chat_approval_unblock_all(&d->approval_q);
                    ChatMsgItem *it = d->msg_list.head;
                    while (it) {
                        if (it->type == CHAT_ITEM_COMMAND &&

                            it->u.cmd.blocked = 0;
                            it->u.cmd.approved = -1;
                        }
                        it = it->next;

                    chat_approval_block_pending_writes(&d->approval_q);
```
